function maxNumber(numbers) {
  var arr = numbers.split(" ").map(Number);
  var largest = arr[0];
  for (var i = 1; i < arr.length; i++) {
    if (arr[i] > largest) {
      largest = arr[i];
    }
  }
  return largest;
}

console.log(maxNumber("1 2 4 5"));
console.log(maxNumber("5 2 7 1 6"));

	let sum = 0,
		productCount = 0,

		if (data[i].name ='Product'){
